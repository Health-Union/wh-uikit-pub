import { Playground, Props } from "docz";
import { FormLabel } from "@material-ui/core";
import moment from "moment";

import { DisplayDate } from "./DisplayDate";

# DisplayDate

Shows different date format e.g. 'YYYY-MM-DD, M/D, MMMM Do, YYYY'

## Props

<Props of={DisplayDate} />

## Examples

### Default Compact format

<Playground>
  <div>
    <FormLabel>Compact Format: </FormLabel>
    <DisplayDate date={moment().format("YYYY-MM-DD")} />
  </div>
</Playground>

### Short Date Format

<Playground>
  <div>
    <FormLabel>Short Format: </FormLabel>
    <DisplayDate
      date={moment().format("YYYY-MM-DD")}
      format={DisplayDate.formats.short}
    />
  </div>
</Playground>
