import { Playground, Props } from "docz";
import moment from "moment";
import { Box, FormLabel } from "@material-ui/core";

import { RelativeTime } from "./RelativeTime";

# RelativeTime

Shows relative time e.g. 'Just Now or 'A few months ago' from current time.

## Props

<Props of={RelativeTime} />

## Examples

### Just Now

<Playground>
  <div>
    <FormLabel>
      Date:{" "}
      {moment()
        .subtract(1, "minutes")
        .format("YYYY-MM-DD hh:mm:ss")}
      :{" "}
    </FormLabel>
    <RelativeTime
      date={moment()
        .subtract(1, "minutes")
        .toISOString()}
    />
  </div>
</Playground>
